CREATE TABLE `tomegkozlekedes`. `erint` ( 
erkezesiIdoPerc INT NOT NULL,
erintSorszam INT AUTO_INCREMENT NOT NULL,
allomasSorszam INT,
vonalSorszam INT,
PRIMARY KEY (erintSorszam),
FOREIGN KEY (jaratSorszam) REFERENCES jarat(jaratSorszam)
ON DELETE CASCADE ON UPDATE CASCADE,
FOREIGN KEY (vonalSorszam) REFERENCES vonal(vonalSorszam)
ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE = InnoDB;
